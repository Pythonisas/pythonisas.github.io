** Contexto (sin humo)
Vamos a construir un *mini-sistema* que, a partir de características simples de una vivienda, estime su precio.

Esto es *Machine Learning* en modo realista:
- no buscamos “la fórmula perfecta”,
- buscamos un modelo que *aproxime bien* y que sepamos *evaluar*.

** Objetivo
En esta práctica vais a:
1. Cargar un dataset pequeño (CSV) de viviendas.
2. Preparar los datos (limpieza + columnas).
3. Entrenar *dos modelos*:
   - =LinearRegression= (modelo base, rápido y fácil de interpretar)
   - =RandomForestRegressor= (modelo más potente, no lineal)
4. Evaluarlos con métricas de regresión (=MAE= y =RMSE=).
5. Hacer predicciones desde la terminal.

** Audiencia / forma de trabajo
- Trabajo en *tríos*.
- Entregable: un repo/carpeta con scripts ejecutables en terminal.
- Se valora claridad, orden y que el programa no “reviente” al primer error.

** Dataset sugerido (pequeño y entendible)
Usaremos un CSV con columnas típicas (inventadas o semi-realistas):

- =m2= (metros cuadrados) → número
- =habitaciones= → entero
- =banos= → entero
- =zona= → categoría (por ejemplo: =centro=, =sur=, =norte=, =este=, =oeste=)
- =antiguedad= → años
- =precio= → *target* (lo que predecimos)

** Flujo del proyecto (lo importante)
1. =train.py=:
   - Lee =data/viviendas.csv=
   - Separa =X= (features) y =y= (=precio=)
   - Crea un pipeline de preprocesado:
     - Numéricas: *passthrough* (o escalado opcional)
     - Categóricas: OneHotEncoding para =zona=
   - Entrena 2 modelos
   - Mide MAE y RMSE
   - Muestra resultados y decide “ganador”
   - (BONUS) Guarda el modelo ganador en =models/model.joblib=

2. =predict.py=:
   - Carga el modelo guardado
   - Pide por terminal los datos de una vivienda
   - Predice y muestra el precio estimado

** Importante: evaluación (métricas)
- =MAE= (Mean Absolute Error): “de media, ¿cuánto me equivoco en euros?”
- =RMSE= (Root Mean Squared Error): penaliza más los errores grandes.

#!/usr/bin/env python3
"""
Práctica ML 1 — entrenamiento

Uso:
  python3 src/train.py --data data/viviendas.csv

Objetivo:
- Entrenar 2 modelos de regresión
- Evaluarlos (MAE, RMSE)
- Elegir el mejor
- (BONUS) Guardar el modelo ganador
"""

from __future__ import annotations

import argparse
import json
from pathlib import Path

import pandas as pd

from sklearn.compose import ColumnTransformer
from sklearn.model_selection import train_test_split
from sklearn.pipeline import Pipeline

from sklearn.metrics import mean_absolute_error, mean_squared_error

from sklearn.linear_model import LinearRegression
from sklearn.ensemble import RandomForestRegressor

import joblib

from src.utils import ensure_dirs, MODELS_DIR
from src.features import FEATURES, TARGET, NUMERIC, CATEGORICAL


def load_data(csv_path: Path) -> pd.DataFrame:
    df = pd.read_csv(csv_path)

    # Validaciones mínimas
    missing = [c for c in (FEATURES + [TARGET]) if c not in df.columns]
    if missing:
        raise ValueError(f"Faltan columnas en el CSV: {missing}")

    # Limpieza mínima: quitar filas con nulos en columnas clave
    df = df.dropna(subset=FEATURES + [TARGET])

    return df


def make_preprocessor() -> ColumnTransformer:
    """
    Preprocesado:
    - numéricas: passthrough
    - categóricas: OneHot
    """
    from sklearn.preprocessing import OneHotEncoder

    preprocessor = ColumnTransformer(
        transformers=[
            ("num", "passthrough", ___),              # NUMERIC
            ("cat", OneHotEncoder(handle_unknown="ignore"), ___),  # CATEGORICAL
        ]
    )
    return preprocessor


def eval_regression(y_true, y_pred) -> dict:
    mae = mean_absolute_error(y_true, y_pred)
    rmse = mean_squared_error(y_true, y_pred, squared=False)  # RMSE
    return {"mae": float(mae), "rmse": float(rmse)}


def train_and_eval(model, X_train, X_test, y_train, y_test, preprocessor) -> dict:
    pipe = Pipeline(steps=[
        ("prep", preprocessor),
        ("model", model),
    ])

    pipe.fit(X_train, y_train)

    pred = pipe.predict(X_test)
    metrics = eval_regression(y_test, pred)

    return {"pipeline": pipe, "metrics": metrics}


def main() -> int:
    parser = argparse.ArgumentParser()
    parser.add_argument("--data", required=True, help="Ruta al CSV")
    parser.add_argument("--seed", type=int, default=42)
    parser.add_argument("--save", action="store_true", help="Guardar el mejor modelo en models/")
    args = parser.parse_args()

    csv_path = Path(args.data)
    if not csv_path.exists():
        raise FileNotFoundError(csv_path)

    ensure_dirs()

    df = load_data(csv_path)
    X = df[FEATURES]
    y = df[TARGET]

    X_train, X_test, y_train, y_test = train_test_split(
        X, y, test_size=0.2, random_state=args.seed
    )

    preprocessor = make_preprocessor()

    # Modelo 1
    r1 = train_and_eval(
        model=___,   # LinearRegression()
        X_train=X_train, X_test=X_test,
        y_train=y_train, y_test=y_test,
        preprocessor=preprocessor
    )

    # Modelo 2
    r2 = train_and_eval(
        model=___,   # RandomForestRegressor(...)
        X_train=X_train, X_test=X_test,
        y_train=y_train, y_test=y_test,
        preprocessor=preprocessor
    )

    results = {
        "linear_regression": r1["metrics"],
        "random_forest": r2["metrics"],
    }

    print("=== Resultados (menor es mejor) ===")
    for name, m in results.items():
        print(f"- {name}: MAE={m['mae']:.2f} | RMSE={m['rmse']:.2f}")

    # Elegimos el mejor por RMSE
    best_name = min(results, key=lambda k: results[k]["rmse"])
    best_pipe = r1["pipeline"] if best_name == "linear_regression" else r2["pipeline"]

    print(f"\nMejor modelo (por RMSE): {best_name}")

    if args.save:
        model_path = MODELS_DIR / "model.joblib"
        joblib.dump(best_pipe, model_path)
        print(f"Modelo guardado en: {model_path}")

        meta = {
            "best_model": best_name,
            "metrics": results,
            "features": FEATURES,
            "target": TARGET,
        }
        (MODELS_DIR / "metadata.json").write_text(json.dumps(meta, indent=2), encoding="utf-8")
        print(f"Metadata guardada en: {MODELS_DIR / 'metadata.json'}")

    return 0


if __name__ == "__main__":
    raise SystemExit(main())

#!/usr/bin/env python3
"""
Práctica ML 1 — predicción por terminal

Uso:
  python3 src/predict.py

Requisitos:
- Haber entrenado y guardado el modelo:
  python3 src/train.py --data data/viviendas.csv --save
"""

from __future__ import annotations

from pathlib import Path
import joblib

import pandas as pd

from src.utils import MODELS_DIR
from src.features import FEATURES


def ask_int(prompt: str) -> int:
    while True:
        try:
            return int(input(prompt).strip())
        except ValueError:
            print("Introduce un número entero válido.")


def ask_float(prompt: str) -> float:
    while True:
        try:
            return float(input(prompt).strip().replace(",", "."))
        except ValueError:
            print("Introduce un número válido (puede tener decimales).")


def ask_zona() -> str:
    # Simple: dejamos que escriban una zona y el OneHotEncoder
    # manejará desconocidas con handle_unknown="ignore".
    return input("Zona (centro/norte/sur/este/oeste): ").strip().lower()


def main() -> int:
    model_path = MODELS_DIR / "model.joblib"
    if not model_path.exists():
        raise FileNotFoundError(
            "No encuentro models/model.joblib. Entrena y guarda primero con: "
            "python3 src/train.py --data data/viviendas.csv --save"
        )

    model = joblib.load(model_path)

    print("Introduce los datos de la vivienda:")
    m2 = ask_float("m2: ")
    hab = ask_int("habitaciones: ")
    banos = ask_int("banos: ")
    zona = ask_zona()
    antig = ask_int("antiguedad (años): ")

    row = {
        "m2": m2,
        "habitaciones": hab,
        "banos": banos,
        "zona": zona,
        "antiguedad": antig,
    }

    X = pd.DataFrame([row], columns=___)  # FEATURES
    pred = model.predict(X)[0]

    print(f"\nPrecio estimado: {pred:,.2f} €")
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
